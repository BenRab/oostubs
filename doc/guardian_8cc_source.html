52 void guardian (unsigned short slot) {

57  //falls es 32+7 oder 32+15 ist, koennte es ein "unechter"-Interrupt sein
58  if(slot == (32+7)){
59  bFinished = (pic.getISR() & 0x80) == 0;
60  }else if(slot == (32+15)){
61  bFinished = (pic.getISR(true) & 0x80) == 0;
