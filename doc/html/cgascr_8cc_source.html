66 void CGA_Screen::print (const char* string, unsigned int n) {
67  unsigned short x,y;
68  getpos(x,y);
69  for(unsigned int i=0; i<n; i++) {
70  if(x >= COLS || string[i] == '\n')
71  {
72  x = 0;
73  if(y >= ROWS-1)
74  {
75  scrollup();
76  }
77  else
78  y++;
79  if(string[i] == '\n')
80  continue;//sonst wird \n ausgegeben
81  }
82  show(x++, y, string[i], attribut);
83  }
84  setpos(x, y);
85 }
