24 void CGA_Screen::setpos (unsigned short x, unsigned short y) {
25  unsigned short pos;
26  unsigned short high, low;
27 
28  if (x >= COLUMNS || y > ROWS) return;
29 
30  pos = y * COLUMNS + x;
31  low = pos & 0xff;
32  high = (pos >> 8 ) & 0xff;
33 
34  /* Indexregister 14 (Cursor high) an Port INDEX_PORT ausgeben */
35  index_port.outb (14);
36  /* Datenregister an Port DATA_PORT schreiben */
37  data_port.outb (high);
38  /* Indexregister 15 (Cursor low) an Port INDEX_PORT ausgeben */
39  index_port.outb (15);
40  /* Datenregister an Port DATA_PORT schreiben */
41  data_port.outb (low);
42 }

44 void CGA_Screen::getpos (unsigned short& x, unsigned short& y) const{
45  unsigned short pos;
46  unsigned char high, low;
47 
48  /* Indexregister 14 (Cursor high) an Port INDEX_PORT ausgeben */
49  index_port.outb (14);
50  /* Datenregister an Port DATA_PORT auslesen */
51  high = data_port.inb ();
52  /* Indexregister 15 (Cursor low) an Port INDEX_PORT ausgeben */
53  index_port.outb (15);
54  /* Datenregister an Port DATA_PORT auslesen */
55  low = data_port.inb ();
56 
57  pos = ((unsigned short)high) << 8 | low;
58  x = pos % COLUMNS;
59  y = pos / COLUMNS;
60 }
