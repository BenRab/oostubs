00052 void guardian (unsigned short slot) {
00053   //var init
00054   bool bFinished = false;
00055   Gate* gate;
00056   
00057   //falls es 32+7 oder 32+15 ist, koennte es ein "unechter"-Interrupt sein
00058   if(slot == (32+7)){
00059     bFinished = (pic.getISR() & 0x80) == 0;
00060   }else if(slot == (32+15)){
00061     bFinished = (pic.getISR(true) & 0x80) == 0;
00062   }
00063   
00064   if(!bFinished){
00065     //Gate holen
00066     gate = &(plugbox.report(slot));
00067     gate->setInterruptNumber(slot);
00068     //und ausloesen
00069     gate->trigger();
00070   }else{
00071     uiSpuriousInterruptCount++;
00072     kout << "Spurious Interrupt (Nr. " << uiSpuriousInterruptCount << ") " << endl;
00073   }
00074 }
